$$\displaylines{

\left( \dfrac{x}{a} - \dfrac{z}{c} \right) p_{ 2 } = \left( 1+\dfrac{y}{b} \right)q_{2}  \\
\left( \dfrac{x}{a} + \dfrac{z}{c} \right)q_{ 2 } = \left( 1 - \dfrac{y}{b} \right)p_{2}
\end{cases}
}$$
